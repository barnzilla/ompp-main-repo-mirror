{{/*
oms web-service:

Template to run model_mpi.exe on Windows using OpenMPI

Rename this file into:
  mpiModelRun.template.txt
to use it by default to run models on MPI cluster.

Oms web-service process this template as:
  - substitute $arguments
  - combine result into single line
  - skip empty lines
Result of template processing is a command line to run model_mpi.exe

Example of result:
mpiexec -n 16 -wdir models/bin -env key value ../bin/model_mpi.exe -OpenM.LogToFile true

Arguments of template:
struct {
  ModelName string            // model name
  ExePath   string            // path to model exe as: binDir/modelname
  Dir       string            // work directory to run the model
  BinDir    string            // bin directory where model.exe is located
  MpiNp     int               // number of MPI processes
  Args      []string          // model command line arguments
  Env       map[string]string // environment variables to run the model
}
*/}}


mpiexec
-n
{{ .MpiNp}}
{{with .Dir}}
-wdir
{{.}}
{{end}}
{{range $key, $val := .Env}}
-env
{{$key}}
{{$val}}
{{end}}
{{.ExePath}}_mpi.exe
{{range .Args}}
{{.}}
{{end}}
